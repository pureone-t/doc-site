version: '3.3'
services:
  # WebServer config
  webserver:
    build: .
    ports:
     - "80:80" # 80番ポートの解放
    depends_on:
     - redis

  # Redis config
  redis: # redisサービスへの依存
    image: redis:4.0

$ docker-compose up

http://0.0.0.0:80/
